<h3>Form Validation</h3>
<form onsubmit="return validateForm()">
    IP Address: <input type="text" id="ip"><br>
    Date (DD/MM/YYYY): <input type="text" id="date"><br>
    Password: <input type="password" id="pass"><br>
    Confirm: <input type="password" id="cpass"><br>
    <button type="submit">Validate</button>
</form>

<script>
    // Prac 6a: Arithmetic & Factorial
    function doMath() {
        let n = 5; 
        let fact = 1;
        for(let i=1; i<=n; i++) fact *= i;
        
        let msg = "5 is " + (5%2==0 ? "Even" : "Odd") + ". Factorial of 5 is " + fact;
        document.getElementById('output').innerText = msg;
    }

    // Prac 6a: Nested Loop Table
    let tableHTML = "<table class='grid-table' style='width:240px'>";
    for(let r=0; r<8; r++) {
        tableHTML += "<tr>";
        for(let c=0; c<8; c++) {
            tableHTML += "<td></td>";
        }
        tableHTML += "</tr>";
    }
    tableHTML += "</table>";
    document.getElementById('tableContainer').innerHTML = tableHTML;

    // Prac 6b: Date & Greeting
    const now = new Date();
    const hrs = now.getHours();
    let greet = "Good Morning";
    if(hrs >= 12) greet = "Good Afternoon";
    if(hrs >= 17) greet = "Good Evening";
    document.getElementById('greeting').innerText = `${greet}, ${now.toLocaleString()}`;

    // Prac 6b: Validation
    function validateForm() {
        let ip = document.getElementById('ip').value;
        let date = document.getElementById('date').value;
        let p1 = document.getElementById('pass').value;
        let p2 = document.getElementById('cpass').value;

        // Simple Regex Examples
        let ipRegex = /^(?:[0-9]{1,3}\.){3}[0-9]{1,3}$/;
        let dateRegex = /^\d{2}\/\d{2}\/\d{4}$/;

        if (!ipRegex.test(ip)) { alert("Invalid IP"); return false; }
        if (!dateRegex.test(date)) { alert("Invalid Date"); return false; }
        if (p1 !== p2) { alert("Passwords do not match"); return false; }
        
        alert("Form Valid!");
        return false; // Prevent actual submit for demo
    }
</script>
